\section{A5 --- Ensemble Methods}

\begin{frame}{A5 --- Stacking Ensemble (Regression)}
    \textbf{Goal:} Improve A2 regression with a Stacking Regressor.
    \vspace{0.5em}
    \begin{itemize}
        \item \textbf{Champion architecture:}
              \begin{itemize}
                  \item Level 0: Random Forest, Gradient Boosting, Lasso
                  \item Level 1: Lasso with \textit{feature passthrough}
              \end{itemize}
        \item Feature passthrough gave meta-learner access to original features
        \item $p < 0.05$ --- statistically significant improvement over previous prod
    \end{itemize}
    \vspace{0.5em}
    \begin{alertblock}{Key finding}
        Diversifying the base learners (tree-based + linear) was essential to
        break the correlation between level-0 errors.
    \end{alertblock}
\end{frame}

\begin{frame}{A5 --- Stacking Ensemble (Classification)}
    \textbf{Goal:} Improve A3 classification with a Stacking Classifier.
    \vspace{0.5em}
    \begin{itemize}
        \item Level 0: RFC, Extra Trees, Na\"{\i}ve Bayes, Logistic Regression
        \item Level 1: Logistic Regression
        \item Initial ensemble showed \textbf{no significant gain} ($p > 0.05$)
    \end{itemize}
    \vspace{0.5em}
    \textbf{Class Imbalance Strategies Tried}
    \begin{enumerate}
        \item Full oversampling to $\sim$600 samples --- overfitting, worse results
        \item Thresholded oversampling ($<$50 samples) --- marginal F1 improvement
        \item Data pruning (drop classes $<$10 instances) --- cleaner confusion matrix
    \end{enumerate}
\end{frame}
